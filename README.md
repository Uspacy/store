## Install

```sh
npm install @uspacy/store
```

or

```sh
yarn add @uspacy/store
```

## Example

```javascript
import { StoreProvider } from '@uspacy/store';

const App: React.FC = () => {
	return (
		<StoreProvider>
			app
		</StoreProvider>
	);
};

export default App;
```
